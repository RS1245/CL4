Practical 3 — Student Grade Calculation Using Hadoop MapReduce



# STEP 1 — Start Hadoop
start-dfs.sh

# STEP 2 — Check Hadoop
jps

# STEP 3 — Create Practical Folder
mkdir Practical3
cd Practical3

# STEP 4 — Create mapper.py
nano mapper.py

Paste EXACTLY:

#!/usr/bin/env python3

import sys

for line in sys.stdin:

    line = line.strip()

    if line:

        name, marks = line.split(',')

        print(f"{name}\t{marks}")

# STEP 5 — Create reducer.py
nano reducer.py

Paste EXACTLY:

#!/usr/bin/env python3

import sys

for line in sys.stdin:

    name, marks = line.strip().split('\t')

    marks = int(marks)

    if marks >= 90:
        grade = 'A'

    elif marks >= 80:
        grade = 'B'

    elif marks >= 70:
        grade = 'C'

    elif marks >= 60:
        grade = 'D'

    else:
        grade = 'F'

    print(f"{name}\t{grade}")

# STEP 6 — Give Execute Permission
chmod +x mapper.py reducer.py

# STEP 7 — Create Input File
nano students.txt

Write:

Alice,85
Bob,67
Charlie,92
David,74
Eva,58

cat students.txt | python3 mapper.py | python3 reducer.py

# STEP 9 — create Folder to HDFS
hadoop fs -mkdir /input

# STEP 10 — Upload File to HDFS
hadoop fs -put students.txt /input


# STEP 11 — Run Hadoop MapReduce Job
hadoop jar /usr/local/hadoop/share/hadoop/tools/lib/hadoop-streaming-3.3.6.jar \
-files mapper.py,reducer.py \
-mapper mapper.py \
-reducer reducer.py \
-input /input/students.txt \
-output /output_grades

# STEP 12 — View Final Output
hadoop fs -cat /output_grades/part-00000


# Expected Output
    Alice  B
    Bob    D
    Charlie A
    David  C
    Eva    F
